# PROPH - Basketball Recruiting Platform
## Cursor IDE Context File

---

## PROJECT OVERVIEW

Proph is a mobile-first basketball recruiting platform that connects high school players with college coaches through algorithmic matching. Think "Tinder for recruiting" meets "Indeed job alerts" with data-driven player evaluation.

### Core Value Proposition
- **For Players:** Browse personalized recruitment postings, apply with one tap, track application status
- **For Coaches:** Post openings, review applications with algorithmic match scores, connect with qualified players
- **Key Differentiator:** Algorithm predicts player's college level (D1/D2/D3/NAIA) to ensure quality matches

### Target Users
- **Primary:** High school basketball players (ages 15-18) and their parents
- **Secondary:** College coaching staffs (D1/D2/D3/NAIA)
- **Focus:** Underserved D2/D3/NAIA players (the "other 95%")

```typescript
// Format date relative to now
export function formatRelativeDate(date: string): string {
  const now = new Date();
  const then = new Date(date);
  const diffMs = now.getTime() - then.getTime();
  const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24));
  
  if (diffDays === 0) return 'Today';
  if (diffDays === 1) return 'Yesterday';
  if (diffDays < 7) return `${diffDays} days ago`;
  if (diffDays < 30) return `${Math.floor(diffDays / 7)} weeks ago`;
  return `${Math.floor(diffDays / 30)} months ago`;
}

// Calculate urgency for deadlines
export function isDeadlineUrgent(deadline: string): boolean {
  const now = new Date();
  const deadlineDate = new Date(deadline);
  const diffMs = deadlineDate.getTime() - now.getTime();
  const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24));
  return diffDays <= 7;
}

// Format match score as tier
export function getMatchTier(score: number): 'great-fit' | 'good-fit' | 'possible-fit' {
  if (score >= 80) return 'great-fit';
  if (score >= 60) return 'good-fit';
  return 'possible-fit';
}

// Truncate text with ellipsis
export function truncate(text: string, maxLength: number): string {
  if (text.length <= maxLength) return text;
  return text.slice(0, maxLength) + '...';
}
```
